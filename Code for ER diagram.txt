Code for ER diagram

User table
CREATE TABLE `hospital_db`.`user` (
  `user_id` INT NOT NULL AUTO_INCREMENT,
  `username` VARCHAR(45) NOT NULL,
  `password` VARCHAR(45) NOT NULL,
  `role` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`user_id`),
  UNIQUE INDEX `username_UNIQUE` (`username` ASC) VISIBLE);

Doctors table
CREATE TABLE `hospital_db`.`doctors` (
  `doctor_id` INT NOT NULL,
  `name` VARCHAR(45) NOT NULL,
  `specialization` VARCHAR(45) NOT NULL,
  `availability` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`doctor_id`));

Patients table
CREATE TABLE `hospital_db`.`patients` (
  `patients_id` INT NOT NULL AUTO_INCREMENT,
  `name` VARCHAR(45) NOT NULL,
  `age` INT NOT NULL,
  `contact_number` INT NOT NULL,
  PRIMARY KEY (`patients_id`));


Appointment table
CREATE TABLE `hospital_db`.`apoointment` (
  `apoointment_id` INT NOT NULL,
  `patient_id` INT NULL,
  `doctor_id` INT NULL,
  `appointment_date` VARCHAR(45) NOT NULL,
  `status` VARCHAR(45) NULL,
  PRIMARY KEY (`apoointment_id`),
  INDEX `patient_id_idx` (`patient_id` ASC) VISIBLE,
  INDEX `doctor_id_idx` (`doctor_id` ASC) VISIBLE,
  CONSTRAINT `patient_id`
    FOREIGN KEY (`patient_id`)
    REFERENCES `hospital2`.`patients` (`patients_id`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT `doctor_id`
    FOREIGN KEY (`doctor_id`)
    REFERENCES `hospital2`.`doctors` (`doctor_id`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);
